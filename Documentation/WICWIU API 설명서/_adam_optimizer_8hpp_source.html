217  int UpdateParameter(Operator<DTYPE> *pParameter, Tensor<DTYPE> *pFirstMomentum, Tensor<DTYPE> *pFirstVelocity, Tensor<DTYPE> *pUnbiasedMomentum, Tensor<DTYPE> *pUnbiasedVelocity) {
218  Tensor<DTYPE> *trainable_data = pParameter->GetResult();
219  Tensor<DTYPE> *gradient = pParameter->GetGradient();
220 
221  float signed_learning_rate = this->GetOptimizeDirection() * this->GetLearningRate();
222 
223  int capacity = trainable_data->GetCapacity();
224 
225  for (int i = 0; i < capacity; i++) {
226  (*pFirstMomentum)[i] = (m_Beta1 * (*pFirstMomentum)[i]) + ((1.f - m_Beta1) * (*gradient)[i]);
227  (*pFirstVelocity)[i] = (m_Beta2 * (*pFirstVelocity)[i]) + ((1.f - m_Beta2) * ((*gradient)[i] * (*gradient)[i]));
228  (*pUnbiasedMomentum)[i] = (*pFirstMomentum)[i] / (1.f - m_Beta1);
229  (*pUnbiasedVelocity)[i] = (*pFirstVelocity)[i] / (1.f - m_Beta2);
230  (*trainable_data)[i] += ((signed_learning_rate * (*pUnbiasedMomentum)[i]) / (std::sqrt((*pUnbiasedVelocity)[i]) + m_epsilon));
231  }
