119 template<typename DTYPE> int Tensor<DTYPE>::Alloc(Shape *pShape, IsUseTime pAnswer) {
120  #ifdef __DEBUG__
121  std::cout << "Tensor<DTYPE>::Alloc(Shape *pShape, IsUseTime pAnswer)" << '\n';
122  #endif // __DEBUG__
123 
124  if (pShape == NULL) {
125  printf("Receive NULL pointer of Shape class in %s (%s %d)\n", __FUNCTION__, __FILE__, __LINE__);
126  return FALSE;
127  } else {
128  m_aShape = pShape;
129  m_IsUseTime = pAnswer;
130 
131  int rank = pShape->GetRank();
132 
133  int pTime = 1;
134  int pCapacityPerTime = 1;
135 
136  if (m_IsUseTime == UseTime) {
137  pTime = (*pShape)[0];
138 
139  for (int i = 1; i < rank; i++) {
140  pCapacityPerTime *= (*pShape)[i];
141  }
142  } else if (m_IsUseTime == NoUseTime) {
143  for (int i = 0; i < rank; i++) {
144  pCapacityPerTime *= (*pShape)[i];
145  }
146  } else return FALSE;
147 
148  m_aLongArray = new LongArray<DTYPE>(pTime, pCapacityPerTime);
149  }
150 
151  m_Device = CPU;
152 
153  return TRUE;
154 }
